        // BERECHNUNG UND ANZEIGE

        function berechnen() {

            // Trade Republic

            const tr_securities_start = parseFloat(document.getElementById('tr_securities_start').value) || 0;

            const tr_securities_today = parseFloat(document.getElementById('tr_securities_today').value) || 0;

            const tr_cash_today = parseFloat(document.getElementById('tr_cash_today').value) || 0;

            const tr_inflow = parseFloat(document.getElementById('tr_inflow').value) || 0;

            const tr_outflow = parseFloat(document.getElementById('tr_outflow').value) || 0;

            

            const tr_cash_start = tr_cash_today - tr_inflow + tr_outflow;

            const tr_start_capital = tr_securities_start + tr_cash_start;

            const tr_total_today = tr_securities_today + tr_cash_today;

            const tr_perf_euro = tr_total_today - tr_start_capital - tr_inflow + tr_outflow;

            const tr_perf_percent = tr_start_capital + tr_inflow > 0 ? (tr_perf_euro / (tr_start_capital + tr_inflow)) * 100 : 0;


            // IBKR

            const ibkr_start = parseFloat(document.getElementById('ibkr_start').value) || 0;

            const ibkr_end = parseFloat(document.getElementById('ibkr_end').value) || 0;

            const ibkr_inflow = parseFloat(document.getElementById('ibkr_inflow').value) || 0;

            const ibkr_outflow = parseFloat(document.getElementById('ibkr_outflow').value) || 0;

            

            const ibkr_perf_euro = ibkr_end - ibkr_start - ibkr_inflow + ibkr_outflow;

            const ibkr_perf_percent = ibkr_start + ibkr_inflow > 0 ? (ibkr_perf_euro / (ibkr_start + ibkr_inflow)) * 100 : 0;


            // Bitvavo

            const bv_start = parseFloat(document.getElementById('bv_start').value) || 0;

            const bv_end = parseFloat(document.getElementById('bv_end').value) || 0;

            const bv_inflow = parseFloat(document.getElementById('bv_inflow').value) || 0;

            const bv_outflow = parseFloat(document.getElementById('bv_outflow').value) || 0;

            

            const bv_perf_euro = bv_end - bv_start - bv_inflow + bv_outflow;

            const bv_perf_percent = bv_start + bv_inflow > 0 ? (bv_perf_euro / (bv_start + bv_inflow)) * 100 : 0;


            // ANZEIGE TRADE REPUBLIC

            document.getElementById('tr_start_capital').textContent = formatCurrency(tr_start_capital);

            document.getElementById('tr_total_today').textContent = formatCurrency(tr_total_today);

            displayBrokerPerf('tr', tr_perf_euro, tr_perf_percent);


            // ANZEIGE IBKR

            displayBrokerPerf('ibkr', ibkr_perf_euro, ibkr_perf_percent);


            // ANZEIGE BITVAVO

            displayBrokerPerf('bv', bv_perf_euro, bv_perf_percent);


            // GESAMTÜBERSICHT

            const totalStart = tr_start_capital + ibkr_start + bv_start;

            const totalEnd = tr_total_today + ibkr_end + bv_end;

            const totalInflow = tr_inflow + ibkr_inflow + bv_inflow;

            const totalOutflow = tr_outflow + ibkr_outflow + bv_outflow;

            const totalPerf_euro = totalEnd - totalStart - totalInflow + totalOutflow;

            const totalPerf_percent = totalStart + totalInflow > 0 ? (totalPerf_euro / (totalStart + totalInflow)) * 100 : 0;


            document.getElementById('totalStart').textContent = formatCurrency(totalStart);

            document.getElementById('totalEnd').textContent = formatCurrency(totalEnd);

            document.getElementById('totalInflows').textContent = formatCurrency(totalInflow);

            document.getElementById('totalOutflows').textContent = formatCurrency(totalOutflow);

            document.getElementById('totalPerformanceEuro').textContent = formatCurrency(totalPerf_euro);

            document.getElementById('totalPerformanceEuro').className = 'stat-value ' + (totalPerf_euro >= 0 ? 'positive' : 'negative');

            document.getElementById('totalPerformancePercent').textContent = formatPercent(totalPerf_percent);

            document.getElementById('totalPerformancePercent').className = 'stat-value ' + (totalPerf_percent >= 0 ? 'positive' : 'negative');


            // DIAGRAMME

            updateChartsGesamt(totalStart, totalEnd, totalPerf_euro, totalInflow);

            updateChartBroker('chartIBKR', ibkr_start, ibkr_end, ibkr_inflow, ibkr_outflow, ibkr_perf_euro);

            updateChartBroker('chartBitvavo', bv_start, bv_end, bv_inflow, bv_outflow, bv_perf_euro);

        }

            document.getElementById(id + '_perf_percent').textContent = formatPercent(perf_percent);

            document.getElementById(id + '_perf_percent').className = 'result-value ' + (perf_percent >= 0 ? 'positive' : 'negative');

        }

        // FORMATIERUNG

        function formatCurrency(value) {

            return new Intl.NumberFormat('de-DE', {

                style: 'currency',

                currency: 'EUR'

            }).format(value);

        }


        function formatPercent(value) {

            return value.toLocaleString('de-DE', {

                minimumFractionDigits: 2,

                maximumFractionDigits: 2

            }) + '%';

        }


        // CSV EXPORT

        function exportCSV() {

            const tr_securities_start = parseFloat(document.getElementById('tr_securities_start').value) || 0;

            const tr_securities_today = parseFloat(document.getElementById('tr_securities_today').value) || 0;

            const tr_cash_today = parseFloat(document.getElementById('tr_cash_today').value) || 0;

            const tr_inflow = parseFloat(document.getElementById('tr_inflow').value) || 0;

            const tr_outflow = parseFloat(document.getElementById('tr_outflow').value) || 0;

            const tr_cash_start = tr_cash_today - tr_inflow + tr_outflow;

            const tr_start = tr_securities_start + tr_cash_start;

            const tr_total_today = tr_securities_today + tr_cash_today;


            const ibkr_start = parseFloat(document.getElementById('ibkr_start').value) || 0;

            const ibkr_end = parseFloat(document.getElementById('ibkr_end').value) || 0;

            const ibkr_inflow = parseFloat(document.getElementById('ibkr_inflow').value) || 0;

            const ibkr_outflow = parseFloat(document.getElementById('ibkr_outflow').value) || 0;


            const bv_start = parseFloat(document.getElementById('bv_start').value) || 0;

            const bv_end = parseFloat(document.getElementById('bv_end').value) || 0;

            const bv_inflow = parseFloat(document.getElementById('bv_inflow').value) || 0;

            const bv_outflow = parseFloat(document.getElementById('bv_outflow').value) || 0;


            function calcPerf(start, end, inflow, outflow) {

                const perf_euro = end - start - inflow + outflow;

                const perf_percent = start + inflow > 0 ? ((perf_euro / (start + inflow)) * 100).toFixed(2) : 0;

                return { euro: perf_euro.toFixed(2), percent: perf_percent };

            }


            const tr_perf = calcPerf(tr_start, tr_total_today, tr_inflow, tr_outflow);

            const ibkr_perf = calcPerf(ibkr_start, ibkr_end, ibkr_inflow, ibkr_outflow);

            const bv_perf = calcPerf(bv_start, bv_end, bv_inflow, bv_outflow);


            const total_start = tr_start + ibkr_start + bv_start;

            const total_end = tr_total_today + ibkr_end + bv_end;

            const total_inflow = tr_inflow + ibkr_inflow + bv_inflow;

            const total_outflow = tr_outflow + ibkr_outflow + bv_outflow;

            const total_perf = calcPerf(total_start, total_end, total_inflow, total_outflow);


            const csv = `Broker;Startwert 01.01;Wert heute;Zugänge;Abflüsse;Performance (€);Performance (%)

Trade Republic;${tr_start.toFixed(2)};${tr_total_today.toFixed(2)};${tr_inflow.toFixed(2)};${tr_outflow.toFixed(2)};${tr_perf.euro};${tr_perf.percent}

Interactive Brokers;${ibkr_start.toFixed(2)};${ibkr_end.toFixed(2)};${ibkr_inflow.toFixed(2)};${ibkr_outflow.toFixed(2)};${ibkr_perf.euro};${ibkr_perf.percent}

Bitvavo;${bv_start.toFixed(2)};${bv_end.toFixed(2)};${bv_inflow.toFixed(2)};${bv_outflow.toFixed(2)};${bv_perf.euro};${bv_perf.percent}

GESAMT;${total_start.toFixed(2)};${total_end.toFixed(2)};${total_inflow.toFixed(2)};${total_outflow.toFixed(2)};${total_perf.euro};${total_perf.percent}`;


            const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });

            const link = document.createElement('a');

            link.href = URL.createObjectURL(blob);

            link.download = 'Portfolio_Performance_' + new Date().toISOString().split('T')[0] + '.csv';

            link.click();

        }
